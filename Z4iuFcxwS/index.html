package main

import (
    "fmt"
    "time"
)

var ch = make(chan string, 10)

func download() {

    fmt.Println("Download begins...")
    sleepTime := time.Second * time.Duration(5)
    time.Sleep(sleepTime)
    fmt.Println("Download successful")
    ch <- sleepTime.String()
}

func main() {
    fmt.Println("Main begins...")
    go download()
    go download()
    go download()
    for i := 0; i < 3; i++ {
        msg := <-ch
        fmt.Println(msg)
    }
    fmt.Println("Main end")
}
